 118 # Hotspot expects the variable FULL_DEBUG_SYMBOLS=1/0 to control debug symbols

 123   ALT_OBJCOPY:=$(OBJCOPY)
 124 else
 125   FULL_DEBUG_SYMBOLS=0
 126 endif

 132   ZIP_DEBUGINFO_FILES:=0
 133 endif
